\addsec{List of Abbreviations}
\begin{acronym}[MIR] 
%In der Eckigen Klammer muss das längste Akronym stehen. 
%Es wird dann die gesamte weitere Tabelle danach ausgerichtet.
\acro{mir}[MIR]{Mid-level Intermediate Representation: Represents a simplified form of Rust code, represents a control-flow graph, produced by the compiler and used for checks like the borrow checker \cite{rust_foundation_mir_nodate}. Acts often as the base-layer for verification tools. Requires the usage of the nightly (unstable) version of the Rust compiler.}
\acro{hir}[HIR]{High-level Intermediate Representation: Represents a compiler-friendly form of the abstract syntax tree after parsing \cite{rust_foundation_hir_2023}. Requires the usage of the nightly (unstable) version of the Rust compiler.}
\end{acronym}
